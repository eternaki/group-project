## Model Architecture

```python
import torch.nn as nn

class EmotionClassifier(nn.Module):
    def __init__(self, num_keypoints=20, num_classes=6):
        super().__init__()
        input_size = num_keypoints * 3  # x, y, visibility

        self.model = nn.Sequential(
            nn.Linear(input_size, 128),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(128, 64),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(64, num_classes)
        )

    def forward(self, x):
        return self.model(x)
```
