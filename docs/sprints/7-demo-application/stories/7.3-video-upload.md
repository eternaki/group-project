## Implementation

```python
video_file = st.file_uploader(
    "Upload Video",
    type=["mp4"],
    help="Upload a short video (max 30 sec)"
)

if video_file:
    # Save to temp file
    with tempfile.NamedTemporaryFile(suffix=".mp4", delete=False) as f:
        f.write(video_file.read())
        video_path = f.name

    # Get video info
    from packages.pipeline.video import VideoProcessor
    vp = VideoProcessor()
    info = vp.get_video_info(Path(video_path))

    st.write(f"Duration: {info['duration']:.1f}s")
    st.write(f"Resolution: {info['width']}x{info['height']}")

    if info['duration'] > 30:
        st.error("Video too long. Max 30 seconds.")
```
