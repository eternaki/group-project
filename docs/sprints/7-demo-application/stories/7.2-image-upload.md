    "Upload Image",
    type=["jpg", "jpeg", "png"],
    help="Upload a dog image (max 10MB)"
)

if uploaded_file:
    # Check file size
    if uploaded_file.size > 10 * 1024 * 1024:
        st.error("File too large. Max 10MB.")
    else:
        # Load and display
        image = Image.open(uploaded_file)
        st.image(image, caption="Uploaded Image", use_column_width=True)
