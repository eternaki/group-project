# apps/demo/app.py

import streamlit as st
from pathlib import Path

st.set_page_config(
    page_title="Dog FACS Demo",
    page_icon="üêï",
    layout="wide"
)

@st.cache_resource
def load_pipeline():
    from packages.pipeline import InferencePipeline, PipelineConfig

    config = PipelineConfig(
        bbox_weights=Path("models/bbox.pt"),
        breed_weights=Path("models/breed.pt"),
        keypoints_weights=Path("models/keypoints.pt"),
        emotion_weights=Path("models/emotion.pt")
    )
    pipeline = InferencePipeline(config)
    pipeline.load()
    return pipeline

def main():
    st.title("Dog FACS - Emotion Analysis")

    # Sidebar
    with st.sidebar:
        st.header("Settings")
        confidence = st.slider(
            "Confidence Threshold",
            0.0, 1.0, 0.5
        )

    # Main content
    st.write("Upload an image or video to analyze dog emotions.")

if __name__ == "__main__":
    main()
